JupyterLab Launcher Debug Helper

CSS Debug Commands (Run in Browser Console)

// 1. Check if launcher body is scrolled document.querySelector('.jp-Launcher-body').scrollTop document.querySelector('.jp-Launcher-body').scrollLeft // 2. Reset launcher scroll position document.querySelector('.jp-Launcher-body').scrollTop = 0; document.querySelector('.jp-Launcher-body').scrollLeft = 0; // 3. Temporarily remove overflow hidden on main document.querySelector('#main').style.overflow = 'visible'; // 4. Check Firefox launcher card position const firefoxCards = document.querySelectorAll('.jp-LauncherCard-icon[title*="Firefox"], .jp-LauncherCard[title*="Firefox"]'); console.log('Firefox cards found:', firefoxCards.length); firefoxCards.forEach((card, i) => { const rect = card.getBoundingClientRect(); console.log(`Firefox card ${i}:`, rect); // Highlight the card card.style.border = '3px solid red'; card.style.backgroundColor = 'yellow'; }); // 5. Force launcher cards to be visible document.querySelectorAll('.jp-LauncherCard-icon').forEach(card => { card.style.position = 'relative'; card.style.zIndex = '9999'; }); // 6. Check all launcher cards in "Other" category const otherSection = Array.from(document.querySelectorAll('.jp-Launcher-section')).find(s => s.querySelector('.jp-Launcher-sectionHeader')?.textContent?.includes('Other') ); if (otherSection) { console.log('Other section found:', otherSection); const cards = otherSection.querySelectorAll('.jp-LauncherCard'); console.log('Cards in Other section:', cards.length); cards.forEach((card, i) => { const rect = card.getBoundingClientRect(); console.log(`Other card ${i}:`, card.getAttribute('title'), rect); }); } else { console.log('Other section NOT found'); }

Force Launcher Visibility CSS

/* Add this CSS in browser dev tools */ .jp-Launcher-body { overflow: visible !important; } #main { overflow: visible !important; } .jp-LauncherCard-icon[title*="Firefox"] { border: 3px solid red !important; background: yellow !important; position: relative !important; z-index: 9999 !important; } /* Force all launcher content to be visible */ .jp-Launcher-content { min-height: auto !important; position: relative !important; }
